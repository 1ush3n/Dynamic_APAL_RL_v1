## 一、项目级统一基础层改造（所有实验的兼容性基石）
### 1.1 统一环境调用接口
所有实验脚本（Baselines/DQN/Basic PPO/Heuristic/消融/泛化）必须调用**同一版 `environment.py`**，且适配环境的 `reset()`/`step()` 接口规范，避免各脚本私自修改环境逻辑。

#### 修改点：
- 在项目根目录创建 `env_wrapper.py`，封装环境初始化/重置/步执行的统一逻辑，所有实验脚本通过该 wrapper 调用环境，避免重复代码和接口不一致；
- 环境的 `seed`/`data_path` 从命令行参数统一传入，而非硬编码；
- 标准化环境返回值（`state, reward, done, info`），确保所有算法能正确解析。

#### 代码实现（env_wrapper.py）：
```python
# 项目根目录/env_wrapper.py
import os
import numpy as np
import torch
from environment import AirLineEnv_Graph

def init_env(args, seed=None):
    """
    统一的环境初始化函数，所有实验脚本必须通过此函数创建环境
    Args:
        args: 命令行参数（需包含 data_path）
        seed: 随机种子（优先用 args.seed，无则用全局配置）
    Returns:
        env: 初始化后的 AirLineEnv_Graph 实例
    """
    # 统一数据路径处理
    data_path = getattr(args, 'data_path', 'data/100.csv')
    if not os.path.isabs(data_path):
        data_path = os.path.join(os.getcwd(), data_path)
    if not os.path.exists(data_path):
        raise FileNotFoundError(f"数据文件不存在: {data_path}")
    
    # 统一 Seed 处理
    env_seed = seed if seed is not None else getattr(args, 'seed', 42)
    np.random.seed(env_seed)
    torch.manual_seed(env_seed) if torch.cuda.is_available() else None
    
    # 创建环境
    env = AirLineEnv_Graph(
        data_path=data_path,
        seed=env_seed,
        render_mode=None
    )
    return env

def standardize_env_step(env, action):
    """
    标准化 env.step() 调用，确保所有算法拿到一致的返回值格式
    """
    try:
        state, reward, done, info = env.step(action)
        # 统一 reward 类型（避免部分场景返回 tensor）
        reward = float(reward)
        # 统一 done 为 bool 类型
        done = bool(done)
        return state, reward, done, info
    except Exception as e:
        raise RuntimeError(f"环境 step 执行失败: {str(e)}") from e

def standardize_env_reset(env, randomize_duration=False):
    """
    标准化 env.reset() 调用
    """
    state = env.reset(randomize_duration=randomize_duration)
    # 确保 state 为 numpy/tensor 格式，无空值
    if state is None:
        raise ValueError("环境 reset 后返回空状态")
    return state
```

### 1.3 统一日志与结果归档
所有实验脚本按统一规则将日志、结果、模型权重保存到 `results/` 子目录，避免文件混乱。

#### 修改点：
- 创建 `utils/logger.py` 统一日志记录（控制台输出 + 文件保存）；
- 创建 `utils/result_saver.py` 统一结果归档（CSV/JSON格式，包含实验参数、指标、耗时）；
- 每个实验自动生成唯一子目录（如 `results/spt_100csv_20240520_143022`），包含日志、结果、配置文件。

#### 代码实现（utils/logger.py）：
```python
# 项目根目录/utils/logger.py
import os
import time
import logging
from datetime import datetime

def init_logger(args, experiment_name):
    """
    初始化统一日志器
    Args:
        args: 命令行参数（包含 result_dir）
        experiment_name: 实验名称（如 spt/dqn/basic_ppo/ablation_no_gat）
    Returns:
        logger: 配置好的日志器
        exp_dir: 实验结果目录路径
    """
    # 生成唯一实验目录
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    data_basename = os.path.splitext(os.path.basename(args.data_path))[0]
    exp_dir = os.path.join(args.result_dir, f"{experiment_name}_{data_basename}_{timestamp}")
    os.makedirs(exp_dir, exist_ok=True)
    
    # 配置日志
    log_file = os.path.join(exp_dir, f"{experiment_name}.log")
    logger = logging.getLogger(experiment_name)
    logger.setLevel(logging.INFO)
    logger.handlers.clear()  # 避免重复 Handler
    
    # 控制台 Handler
    console_handler = logging.StreamHandler()
    console_handler.setFormatter(logging.Formatter('%(asctime)s - %(levelname)s - %(message)s'))
    # 文件 Handler
    file_handler = logging.FileHandler(log_file, encoding='utf-8')
    file_handler.setFormatter(logging.Formatter('%(asctime)s - %(levelname)s - %(message)s'))
    
    logger.addHandler(console_handler)
    logger.addHandler(file_handler)
    
    # 记录实验参数
    logger.info(f"实验参数: {vars(args)}")
    logger.info(f"实验目录: {exp_dir}")
    logger.info(f"开始时间: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
    
    return logger, exp_dir

def record_experiment_time(logger, start_time):
    """
    记录实验耗时
    """
    end_time = time.time()
    elapsed = end_time - start_time
    hours = int(elapsed // 3600)
    minutes = int((elapsed % 3600) // 60)
    seconds = int(elapsed % 60)
    logger.info(f"实验结束时间: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
    logger.info(f"总耗时: {hours}小时{minutes}分钟{seconds}秒")
    return elapsed
```

```python
import torch

def get_available_device():
    """
    获取可用设备（优先GPU，无则CPU），并检查GPU可用性
    """
    if torch.cuda.is_available():
        device = torch.device("cuda:0")
        # 检查GPU显存是否充足（简单判断）
        free_mem = torch.cuda.mem_get_info(device)[0] / 1024 / 1024 / 1024  # GB
        if free_mem < 1.0:
            print(f"警告：GPU显存不足（剩余{free_mem:.1f}GB），将切换到CPU")
            device = torch.device("cpu")
        print(f"使用设备: {device} (GPU显存剩余: {free_mem:.1f}GB)")
    else:
        device = torch.device("cpu")
        print("使用设备: CPU（无可用GPU）")
    return device

def clear_torch_cache():
    """
    安全清理Torch缓存，避免显存泄漏
    """
    if torch.cuda.is_available():
        torch.cuda.empty_cache()
        torch.cuda.synchronize()
```

## 二、Baselines 模块专项修复（heuristic/DQN/Basic PPO）
### 2.1 Heuristic (SPT) 基线（baselines/heuristic/run_spt.py）
#### 核心问题：
- 未调用统一环境接口，可能与主环境逻辑不一致；
- 无结果归档、日志记录；
- 随机指派工人逻辑可能触发死锁，无容错。

#### 修改计划：
```python
# baselines/heuristic/run_spt.py
import os
import time
import numpy as np
import pandas as pd
from args_parser import get_base_parser
from env_wrapper import init_env, standardize_env_reset, standardize_env_step
from utils.logger import init_logger, record_experiment_time
from utils.device_utils import clear_torch_cache

def spt_policy(env):
    """
    SPT策略：优先选择耗时最短的就绪任务，随机指派符合条件的工人
    增加死锁容错逻辑
    """
    # 获取就绪任务（status=1）
    ready_tasks = np.where(env.task_status == 1)[0]
    if len(ready_tasks) == 0:
        return None  # 无就绪任务
    
    # 按耗时升序排序（SPT核心）
    task_durations = env.task_static_feat[ready_tasks, 0].numpy()
    sorted_task_idx = np.argsort(task_durations)
    selected_task = ready_tasks[sorted_task_idx[0]]
    
    # 选择站位（优先固定站位，无则随机）
    fixed_station = env.fixed_stations[selected_task]
    if fixed_station != -1 and 0 <= fixed_station < env.num_stations:
        selected_station = fixed_station
    else:
        selected_station = np.random.choice(env.num_stations)
    
    # 选择符合技能的工人（避免死锁）
    task_skill = int(env.task_static_feat[selected_task, 1].item())
    qualified_workers = np.where(env.worker_skill_matrix[:, task_skill] > 0)[0]
    if len(qualified_workers) == 0:
        # 容错：无合格工人时强制指派第一个工人（兜底）
        qualified_workers = [0]
    # 随机选工人（按任务需求人数）
    num_workers = int(env.task_static_feat[selected_task, 2].item())
    selected_workers = np.random.choice(qualified_workers, size=min(num_workers, len(qualified_workers)), replace=False)
    
    # 构造动作（适配环境action_space）
    action = (selected_task, selected_station, selected_workers[0])  # 取第一个工人作为leader
    return action

def run_spt(args):
    # 初始化日志
    logger, exp_dir = init_logger(args, "spt_heuristic")
    start_time = time.time()
    
    try:
        # 初始化环境（统一接口）
        env = init_env(args, seed=args.seed)
        
        # 实验指标初始化
        total_makespan = []
        total_reward = []
        total_duration = []
        
        # 运行SPT策略（默认10轮，保证统计稳定性）
        num_runs = getattr(args, 'num_runs', 10)
        for run in range(num_runs):
            logger.info(f"SPT策略运行轮次: {run+1}/{num_runs}")
            state = standardize_env_reset(env)
            done = False
            ep_reward = 0
            step_count = 0
            max_steps = env.num_tasks * 2  # 防止无限循环
            
            while not done and step_count < max_steps:
                step_count += 1
                action = spt_policy(env)
                if action is None:
                    logger.warning(f"轮次{run+1}：无就绪任务，终止本轮")
                    break
                # 标准化step调用
                state, reward, done, info = standardize_env_step(env, action)
                ep_reward += reward
            
            # 记录指标
            makespan = np.max(env.station_wall_clock) if not done else 99999
            total_makespan.append(makespan)
            total_reward.append(ep_reward)
            total_duration.append(time.time() - start_time)
            logger.info(f"轮次{run+1} - Makespan: {makespan:.2f}, 总奖励: {ep_reward:.2f}")
        
        # 结果归档
        results = pd.DataFrame({
            'run': range(1, num_runs+1),
            'makespan': total_makespan,
            'total_reward': total_reward,
            'duration': total_duration
        })
        results['avg_makespan'] = np.mean(total_makespan)
        results['avg_reward'] = np.mean(total_reward)
        results.to_csv(os.path.join(exp_dir, "spt_results.csv"), index=False)
        logger.info(f"平均Makespan: {np.mean(total_makespan):.2f}, 平均奖励: {np.mean(total_reward):.2f}")
        
    except Exception as e:
        logger.error(f"SPT实验执行失败: {str(e)}", exc_info=True)
        raise
    finally:
        # 清理资源
        record_experiment_time(logger, start_time)
        clear_torch_cache()

if __name__ == "__main__":
    parser = get_base_parser()
    parser.add_argument('--num_runs', type=int, default=10, help="SPT策略运行轮次")
    args = parser.parse_args()
    run_spt(args)
```

### 2.2 DQN 基线（baselines/dqn/train_dqn.py）
#### 核心问题：
- 动作空间适配问题（环境是MultiDiscrete，DQN输出维度需匹配）；
- 未处理环境返回值标准化；
- 大数据集（3000.csv）下内存溢出，无分批/显存管理；
- 无日志、结果归档。

#### 修改计划：
```python
# baselines/dqn/train_dqn.py
import os
import time
import torch
import torch.nn as nn
import torch.optim as optim
import numpy as np
import pandas as pd
from collections import deque
from args_parser import get_dqn_parser
from env_wrapper import init_env, standardize_env_reset, standardize_env_step
from utils.logger import init_logger, record_experiment_time
from utils.device_utils import get_available_device, clear_torch_cache

# DQN网络（适配环境MultiDiscrete动作空间）
class DQN(nn.Module):
    def __init__(self, state_dim, action_dim_list, hidden_dim=128):
        super(DQN, self).__init__()
        # 共享特征层
        self.fc1 = nn.Linear(state_dim, hidden_dim)
        self.fc2 = nn.Linear(hidden_dim, hidden_dim)
        # 动作分支（适配MultiDiscrete: [task, station, worker_leader]）
        self.task_head = nn.Linear(hidden_dim, action_dim_list[0])
        self.station_head = nn.Linear(hidden_dim, action_dim_list[1])
        self.worker_head = nn.Linear(hidden_dim, action_dim_list[2])
        
    def forward(self, x):
        x = torch.relu(self.fc1(x))
        x = torch.relu(self.fc2(x))
        task_logits = self.task_head(x)
        station_logits = self.station_head(x)
        worker_logits = self.worker_head(x)
        return task_logits, station_logits, worker_logits

class DQNAgent:
    def __init__(self, state_dim, action_dim_list, args, device):
        self.device = device
        self.gamma = args.gamma
        self.epsilon = args.epsilon
        self.epsilon_min = args.epsilon_min
        self.epsilon_decay = args.epsilon_decay
        self.action_dim_list = action_dim_list
        
        # 初始化网络
        self.model = DQN(state_dim, action_dim_list).to(device)
        self.target_model = DQN(state_dim, action_dim_list).to(device)
        self.target_model.load_state_dict(self.model.state_dict())
        self.optimizer = optim.Adam(self.model.parameters(), lr=1e-3)
        self.loss_fn = nn.MSELoss()
        
        # 经验回放池（限制大小，避免内存溢出）
        self.memory = deque(maxlen=getattr(args, 'memory_size', 10000))
        
    def select_action(self, state):
        """
        带探索的动作选择，适配MultiDiscrete动作空间
        """
        if np.random.rand() <= self.epsilon:
            # 随机动作（探索）
            task = np.random.randint(0, self.action_dim_list[0])
            station = np.random.randint(0, self.action_dim_list[1])
            worker = np.random.randint(0, self.action_dim_list[2])
            return (task, station, worker)
        
        # 贪心动作（利用）
        state = torch.tensor(state, dtype=torch.float32).unsqueeze(0).to(self.device)
        with torch.no_grad():
            task_logits, station_logits, worker_logits = self.model(state)
            task = torch.argmax(task_logits).item()
            station = torch.argmax(station_logits).item()
            worker = torch.argmax(worker_logits).item()
        return (task, station, worker)
    
    def remember(self, state, action, reward, next_state, done):
        self.memory.append((state, action, reward, next_state, done))
    
    def replay(self, batch_size):
        """
        经验回放，分批训练避免内存溢出
        """
        if len(self.memory) < batch_size:
            return 0.0
        
        batch = np.random.choice(len(self.memory), batch_size, replace=False)
        losses = []
        
        for idx in batch:
            state, action, reward, next_state, done = self.memory[idx]
            
            # 转换为tensor
            state = torch.tensor(state, dtype=torch.float32).to(self.device)
            next_state = torch.tensor(next_state, dtype=torch.float32).to(self.device)
            reward = torch.tensor(reward, dtype=torch.float32).to(self.device)
            
            # 当前Q值
            task_logits, station_logits, worker_logits = self.model(state)
            q_current = (task_logits[action[0]] + station_logits[action[1]] + worker_logits[action[2]]) / 3
            
            # 目标Q值
            if done:
                q_target = reward
            else:
                with torch.no_grad():
                    next_task_logits, next_station_logits, next_worker_logits = self.target_model(next_state)
                    next_q = (torch.max(next_task_logits) + torch.max(next_station_logits) + torch.max(next_worker_logits)) / 3
                q_target = reward + self.gamma * next_q
            
            # 计算损失
            loss = self.loss_fn(q_current, q_target)
            losses.append(loss.item())
            
            # 反向传播
            self.optimizer.zero_grad()
            loss.backward()
            self.optimizer.step()
        
        # 衰减探索率
        if self.epsilon > self.epsilon_min:
            self.epsilon *= self.epsilon_decay
        
        return np.mean(losses) if losses else 0.0

def train_dqn(args):
    # 初始化日志
    logger, exp_dir = init_logger(args, "dqn_baseline")
    start_time = time.time()
    
    try:
        # 设备初始化
        device = get_available_device()
        # 环境初始化（统一接口）
        env = init_env(args, seed=args.seed)
        
        # 状态维度适配（根据环境输出的state.shape）
        state = standardize_env_reset(env)
        state_dim = np.prod(state.shape) if isinstance(state, np.ndarray) else 128  # 兜底
        action_dim_list = [env.action_space.nvec[0], env.action_space.nvec[1], env.action_space.nvec[2]]
        
        # 初始化Agent
        agent = DQNAgent(state_dim, action_dim_list, args, device)
        batch_size = getattr(args, 'batch_size', 32)
        
        # 训练指标
        episode_rewards = []
        episode_losses = []
        episode_makespans = []
        
        # 训练循环
        logger.info(f"开始DQN训练，最大轮次: {args.max_episodes}")
        for ep in range(args.max_episodes):
            state = standardize_env_reset(env)
            done = False
            ep_reward = 0
            ep_loss = 0
            step_count = 0
            max_steps = env.num_tasks * 2  # 防止无限循环
            
            while not done and step_count < max_steps:
                step_count += 1
                # 选择动作
                action = agent.select_action(state)
                # 执行动作
                next_state, reward, done, info = standardize_env_step(env, action)
                # 存储经验
                agent.remember(state, action, reward, next_state, done)
                # 累加奖励
                ep_reward += reward
                # 经验回放
                loss = agent.replay(batch_size)
                ep_loss += loss
                # 更新状态
                state = next_state
            
            # 记录指标
            episode_rewards.append(ep_reward)
            episode_losses.append(ep_loss / step_count if step_count > 0 else 0)
            makespan = np.max(env.station_wall_clock) if not done else 99999
            episode_makespans.append(makespan)
            
            # 每10轮打印日志
            if (ep + 1) % 10 == 0:
                avg_reward = np.mean(episode_rewards[-10:])
                avg_loss = np.mean(episode_losses[-10:])
                avg_makespan = np.mean(episode_makespans[-10:])
                logger.info(f"Episode {ep+1}/{args.max_episodes} | 平均奖励: {avg_reward:.2f} | 平均损失: {avg_loss:.4f} | 平均Makespan: {avg_makespan:.2f} | Epsilon: {agent.epsilon:.4f}")
            
            # 每50轮更新目标网络
            if (ep + 1) % 50 == 0:
                agent.target_model.load_state_dict(agent.model.state_dict())
                # 清理缓存
                clear_torch_cache()
        
        # 保存模型
        model_path = os.path.join(exp_dir, "dqn_model.pth")
        torch.save(agent.model.state_dict(), model_path)
        logger.info(f"模型保存至: {model_path}")
        
        # 结果归档
        results = pd.DataFrame({
            'episode': range(1, args.max_episodes+1),
            'reward': episode_rewards,
            'loss': episode_losses,
            'makespan': episode_makespans
        })
        results['avg_reward_10'] = results['reward'].rolling(window=10).mean()
        results['avg_makespan_10'] = results['makespan'].rolling(window=10).mean()
        results.to_csv(os.path.join(exp_dir, "dqn_results.csv"), index=False)
        
    except Exception as e:
        logger.error(f"DQN训练失败: {str(e)}", exc_info=True)
        raise
    finally:
        # 清理资源
        record_experiment_time(logger, start_time)
        clear_torch_cache()

if __name__ == "__main__":
    parser = get_dqn_parser()
    args = parser.parse_args()
    train_dqn(args)
```

#### 核心问题：
- 剥离了图注意力/指针网络，但未适配环境接口；
- 无梯度裁剪、显存管理，大数据集下易崩溃；

#### 修改计划：
```python
# baselines/basic_ppo/train_basic.py
import os
import time
import torch
import torch.nn as nn
import torch.optim as optim
from torch.distributions import Categorical
import numpy as np
import pandas as pd
from args_parser import get_basic_ppo_parser
from env_wrapper import init_env, standardize_env_reset, standardize_env_step
from utils.logger import init_logger, record_experiment_time
from utils.device_utils import get_available_device, clear_torch_cache

# 基础PPO网络（仅MLP，无GAT/指针网络）
class BasicPPO(nn.Module):
    def __init__(self, state_dim, action_dim_list, hidden_dim=256):
        super(BasicPPO, self).__init__()
        # 共享MLP特征层
        self.feature_extractor = nn.Sequential(
            nn.Linear(state_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU()
        )
        # 策略头（适配MultiDiscrete动作空间）
        self.task_policy = nn.Linear(hidden_dim, action_dim_list[0])
        self.station_policy = nn.Linear(hidden_dim, action_dim_list[1])
        self.worker_policy = nn.Linear(hidden_dim, action_dim_list[2])
        # 价值头
        self.value_head = nn.Linear(hidden_dim, 1)
    
    def forward(self, x):
        features = self.feature_extractor(x)
        # 策略输出（logits）
        task_logits = self.task_policy(features)
        station_logits = self.station_policy(features)
        worker_logits = self.worker_policy(features)
        # 价值输出
        value = self.value_head(features)
        return task_logits, station_logits, worker_logits, value

class BasicPPOAgent:
    def __init__(self, state_dim, action_dim_list, args, device):
        self.device = device
        self.lr = args.lr
        self.clip_epsilon = args.clip_epsilon
        self.gamma = getattr(args, 'gamma', 0.99)
        self.lamda = getattr(args, 'lamda', 0.95)
        
        # 初始化网络
        self.model = BasicPPO(state_dim, action_dim_list).to(device)
        self.optimizer = optim.Adam(self.model.parameters(), lr=self.lr)
        
        # 存储轨迹数据（每轮清空，避免内存溢出）
        self.states = []
        self.actions = []
        self.log_probs = []
        self.rewards = []
        self.values = []
        self.dones = []
    
    def select_action(self, state):
        """
        选择动作并记录概率/价值
        """
        state = torch.tensor(state, dtype=torch.float32).unsqueeze(0).to(self.device)
        with torch.no_grad():
            task_logits, station_logits, worker_logits, value = self.model(state)
            
            # 构建分类分布
            task_dist = Categorical(logits=task_logits)
            station_dist = Categorical(logits=station_logits)
            worker_dist = Categorical(logits=worker_logits)
            
            # 采样动作
            task_action = task_dist.sample()
            station_action = station_dist.sample()
            worker_action = worker_dist.sample()
            
            # 记录log prob和value
            log_prob = task_dist.log_prob(task_action) + station_dist.log_prob(station_action) + worker_dist.log_prob(worker_action)
            value = value.item()
        
        action = (task_action.item(), station_action.item(), worker_action.item())
        # 存储轨迹
        self.states.append(state)
        self.actions.append(action)
        self.log_probs.append(log_prob)
        self.values.append(value)
        
        return action
    
    def store_reward(self, reward, done):
        self.rewards.append(reward)
        self.dones.append(done)
    
    def compute_gae(self):
        """
        计算GAE优势函数
        """
        advantages = []
        advantage = 0.0
        next_value = 0.0
        
        for reward, done, value in reversed(list(zip(self.rewards, self.dones, self.values))):
            td_error = reward + self.gamma * next_value * (1 - done) - value
            advantage = td_error + self.gamma * self.lamda * (1 - done) * advantage
            advantages.insert(0, advantage)
            next_value = value
        
        # 计算回报
        returns = np.array(advantages) + np.array(self.values)
        # 标准化优势
        advantages = (advantages - np.mean(advantages)) / (np.std(advantages) + 1e-8)
        return advantages, returns
    
    def update(self, batch_size):
        """
        PPO更新，分批训练避免内存溢出
        """
        advantages, returns = self.compute_gae()
        
        # 转换为tensor
        states = torch.cat(self.states).to(self.device)
        log_probs_old = torch.cat(self.log_probs).to(self.device)
        returns = torch.tensor(returns, dtype=torch.float32).to(self.device)
        advantages = torch.tensor(advantages, dtype=torch.float32).to(self.device)
        
        # 分批更新
        indices = np.arange(len(states))
        np.random.shuffle(indices)
        losses = []
        
        for start in range(0, len(states), batch_size):
            end = start + batch_size
            batch_idx = indices[start:end]
            
            # 前向传播
            task_logits, station_logits, worker_logits, values = self.model(states[batch_idx])
            
            # 重新计算动作概率
            task_dist = Categorical(logits=task_logits)
            station_dist = Categorical(logits=station_logits)
            worker_dist = Categorical(logits=worker_logits)
            
            # 取出批次动作
            batch_actions = [self.actions[i] for i in batch_idx]
            task_actions = torch.tensor([a[0] for a in batch_actions], dtype=torch.int64).to(self.device)
            station_actions = torch.tensor([a[1] for a in batch_actions], dtype=torch.int64).to(self.device)
            worker_actions = torch.tensor([a[2] for a in batch_actions], dtype=torch.int64).to(self.device)
            
            log_probs_new = task_dist.log_prob(task_actions) + station_dist.log_prob(station_actions) + worker_dist.log_prob(worker_actions)
            
            # PPO裁剪
            ratio = torch.exp(log_probs_new - log_probs_old[batch_idx])
            surr1 = ratio * advantages[batch_idx]
            surr2 = torch.clamp(ratio, 1 - self.clip_epsilon, 1 + self.clip_epsilon) * advantages[batch_idx]
            policy_loss = -torch.min(surr1, surr2).mean()
            
            # 价值损失
            value_loss = nn.MSELoss()(values.squeeze(), returns[batch_idx])
            
            # 总损失
            total_loss = policy_loss + 0.5 * value_loss
            
            # 反向传播（梯度裁剪避免爆炸）
            self.optimizer.zero_grad()
            total_loss.backward()
            torch.nn.utils.clip_grad_norm_(self.model.parameters(), max_norm=1.0)
            self.optimizer.step()
            
            losses.append(total_loss.item())
        
        # 清空轨迹
        self.states.clear()
        self.actions.clear()
        self.log_probs.clear()
        self.rewards.clear()
        self.values.clear()
        self.dones.clear()
        
        return np.mean(losses) if losses else 0.0

def train_basic_ppo(args):
```

```python
# train.py 中模型初始化部分
def init_agent(args, state_dim, action_dim, device):
    """
    根据消融开关初始化Agent
    """
    # 核心参数
    agent_kwargs = {
        'state_dim': state_dim,
        'action_dim': action_dim,
        'device': device,
        'lr': args.lr,
        'gamma': args.gamma,
        # 消融开关参数
        'use_gat': not args.ablation_no_gat,
        'use_pointer': not args.ablation_no_pointer,
        'use_hard_mask': not args.ablation_no_mask
    }
    
    # 初始化Agent（根据开关加载不同模块）
    if args.ablation_no_gat and args.ablation_no_pointer:
        # 无GAT+无指针，退化为Basic PPO
        agent = BasicPPOAgent(**agent_kwargs)
    elif args.ablation_no_gat:
        # 无GAT，保留指针网络
        agent = PointerOnlyPPOAgent(**agent_kwargs)
    elif args.ablation_no_pointer:
        # 无指针，保留GAT
        agent = GATOnlyPPOAgent(**agent_kwargs)
    elif args.ablation_no_mask:
        # 无硬掩码，改用软约束
        agent = SoftMaskPPOAgent(**agent_kwargs)
    else:
        # 全功能Agent
        agent = FullPPOAgent(**agent_kwargs)
    
    return agent
```

## 六、最终兼容性保障
1. **代码目录结构强制统一**：所有脚本必须放在指定目录，依赖的 `env_wrapper.py`/`args_parser.py`/`utils/` 必须在项目根目录；
2. **版本兼容**：所有脚本使用相同版本的 `environment.py`，禁止Baselines脚本私自修改环境逻辑；
3. **异常捕获**：所有核心循环（训练/测试）增加 `try/except`，确保单轮错误不导致整体崩溃；
4. **资源清理**：所有脚本在 `finally` 块中清理Torch缓存、关闭日志/文件句柄，避免资源泄漏。

通过以上修改，所有实验脚本（Baselines/消融/泛化测试）将具备高度兼容性，可按使用指南的指令正常运行，且结果归档、日志记录、资源管理统一规范。
